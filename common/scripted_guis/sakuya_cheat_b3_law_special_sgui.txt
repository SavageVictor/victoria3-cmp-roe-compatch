# Sakuya Cheat Menu Pro - Block 3 Law special scripted guis

sakuya_b3_law_caste_hegemony_visible_check = {
	scope = country
	is_shown = {
		is_player = yes
		has_dlc_feature = ip2_content
		OR = {
			has_social_hierarchy = british_indian_caste_system
			any_country = {
				has_variable = sakuya_mark_country_target
				has_social_hierarchy = british_indian_caste_system
			}
		}
	}
}

sakuya_b3_law_sakoku_visible_check = {
	scope = country
	is_shown = {
		is_player = yes
		OR = {
			AND = {
				country_has_primary_culture = cu:japanese
				any_scope_state = {
					state_region = s:STATE_KYUSHU
					is_largest_state_in_region = yes
				}
			}
			any_country = {
				has_variable = sakuya_mark_country_target
				country_has_primary_culture = cu:japanese
				any_scope_state = {
					state_region = s:STATE_KYUSHU
					is_largest_state_in_region = yes
				}
			}
		}
	}
}
